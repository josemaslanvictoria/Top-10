class Aspiradora: def __init__(self, habitacion): self.habitacion = habitacion self.posicion = (0, 0) # Posición inicial self.orientacion = 'derecha' # Orientación inicial def mover(self): x, y = self.posicion if self.orientacion == 'derecha': if x < len(self.habitacion[0]) - 1: x += 1 else: self.orientacion = 'izquierda' y += 1 elif self.orientacion == 'izquierda': if x > 0: x -= 1 else: self.orientacion = 'derecha' y += 1 self.posicion = (x, y) def aspirar(self): x, y = self.posicion if self.habitacion[y][x] == 'sucio': self.habitacion[y][x] = 'limpio' print(f"Aspirando suciedad en {self.posicion}") else: print(f"No hay suciedad en {self.posicion}") def limpiar_habitacion(self): for _ in range(len(self.habitacion) * len(self.habitacion[0])): self.aspirar() self.mover() # Ejemplo de uso habitacion = [ ['sucio', 'limpio', 'sucio'], ['limpio', 'sucio', 'limpio'], ['sucio', 'limpio', 'sucio'] ] aspiradora = Aspiradora(habitacion) aspiradora.limpiar_habitacion() # Mostrar la habitación después de limpiar print("\nEstado final de la habitación:") for fila in habitacion: print(fila)
